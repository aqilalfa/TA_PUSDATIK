## 4.6.1 Arsitektur Session Management

Sistem menggunakan **stateless session architecture** dengan unique session IDs untuk tracking conversation context:

```python
class SessionManager:
    """Manage user sessions dan conversation history."""
    
    def create_session(self, user_id: Optional[str] = None) -> str:
        """
        Create new session dengan unique ID.
        
        Session ID format: UUID v4 (non-traceable untuk privacy)
        """
        session_id = str(uuid.uuid4())
        
        session_record = {
            'session_id': session_id,
            'created_at': datetime.now(timezone.utc),
            'last_activity': datetime.now(timezone.utc),
            'user_id': user_id,  # Optional, untuk internal users
            'conversation_history': [],
            'metadata': {
                'ip_address': get_client_ip(),
                'user_agent': get_user_agent(),
            }
        }
        
        # Store in Redis dengan TTL 24 jam
        redis_client.setex(
            f"session:{session_id}",
            86400,  # 24 hours TTL
            json.dumps(session_record)
        )
        
        return session_id
```

## 4.6.2 LDAP Integration untuk Internal Users

Untuk internal LSPro users, sistem support LDAP authentication:

```python
class LDAPAuthProvider:
    """
    LDAP authentication untuk internal LSPro staff.
    Integrasi dengan BSSN LDAP directory.
    """
    
    def __init__(
        self,
        ldap_server: str,
        ldap_port: int = 389,
        base_dn: str = "dc=bssn,dc=go,dc=id"
    ):
        self.server = Server(ldap_server, port=ldap_port, get_info=ALL)
        self.base_dn = base_dn
    
    async def authenticate(self, username: str, password: str) -> Tuple[bool, Dict]:
        """
        Authenticate user terhadap LDAP server.
        
        Returns: (success: bool, user_info: Dict)
        """
        
        try:
            conn = Connection(
                self.server,
                user=f"uid={username},{self.base_dn}",
                password=password,
                auto_bind=True
            )
            
            # Get user attributes
            conn.search(
                search_base=self.base_dn,
                search_filter=f'(uid={username})',
                attributes=['cn', 'mail', 'memberOf']
            )
            
            if not conn.entries:
                return False, {}
            
            user_data = conn.entries[0]
            
            # Extract role dari LDAP groups
            roles = self._extract_roles_from_groups(user_data.memberOf)
            
            return True, {
                'username': username,
                'full_name': user_data.cn[0],
                'email': user_data.mail[0],
                'roles': roles,
                'ldap_entry': str(user_data.entry_dn)
            }
            
        except Exception as e:
            logger.warning(f"LDAP authentication failed for {username}: {e}")
            return False, {}
    
    def _extract_roles_from_groups(self, member_of: List[str]) -> List[str]:
        """Extract role dari LDAP group membership."""
        
        group_role_mapping = {
            'cn=lspro-admin': 'admin',
            'cn=lspro-staff': 'staff',
            'cn=lspro-viewer': 'viewer',
        }
        
        roles = []
        for group in member_of:
            for group_pattern, role in group_role_mapping.items():
                if group_pattern in group.lower():
                    roles.append(role)
                    break
        
        return roles or ['viewer']  # Default role
```

## 4.6.3 API Rate Limiting dan DDoS Protection

Sistem melindungi dari abuse dengan rate limiting:

```python
class RateLimitMiddleware:
    """
    Rate limiting untuk API endpoints.
    """
    
    # Rate limit configuration (per-session)
    RATE_LIMITS = {
        'query': (30, 60),          # 30 queries per 60 seconds
        'document_upload': (5, 3600), # 5 uploads per hour
        'session_create': (10, 3600), # 10 sessions per hour
    }
    
    async def __call__(
        self,
        request: Request,
        call_next,
        endpoint: str
    ):
        """Check rate limit before processing request."""
        
        session_id = request.headers.get('X-Session-ID', 'anonymous')
        
        # Get rate limit for this endpoint
        if endpoint not in self.RATE_LIMITS:
            return await call_next(request)
        
        limit, window = self.RATE_LIMITS[endpoint]
        
        # Redis key untuk rate limit tracking
        key = f"ratelimit:{session_id}:{endpoint}"
        
        # Increment counter
        current_count = redis_client.incr(key)
        
        if current_count == 1:
            # First request, set TTL
            redis_client.expire(key, window)
        
        # Check limit
        if current_count > limit:
            logger.warning(
                f"Rate limit exceeded for {session_id} on {endpoint}: "
                f"{current_count}/{limit}"
            )
            
            return JSONResponse(
                status_code=429,
                content={
                    "error": "Rate limit exceeded. Silakan coba lagi nanti."
                }
            )
        
        # Add remaining quota to response headers
        response = await call_next(request)
        response.headers["X-RateLimit-Remaining"] = str(limit - current_count)
        response.headers["X-RateLimit-Reset"] = str(window)
        
        return response
```
